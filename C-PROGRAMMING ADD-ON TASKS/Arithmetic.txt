#include <stdio.h>

int main() {
    
    int a;
    int b;
    
    printf("Enter a Value for a : ");
    scanf("%d",&a);
    printf("Enter a Value for b : ");
    scanf("%d",&b);
    
    printf("\na+b = %d  ",a+b);
    printf("\na-b = %d  ",a-b);
    printf("\na*b = %d  ",a*b);
    printf("\na/b = %d  ",a/b);
    

    return 0;
}
